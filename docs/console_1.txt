SELECT
    s.polica,
    sh.name AS shipper_name,
    r.name AS receiver_name,
    c1.name AS origin_country,
    c2.name AS destination_country,
    pt.type_name AS package_type,
    sm.mode_name AS shipping_mode,
    s.actual_weight,
    s.volumetric_weight,
    s.content,
    s.created_at
FROM shipments s
         JOIN shippers sh ON s.shipper_id = sh.id
         JOIN receivers r ON s.receiver_id = r.id
         JOIN countries c1 ON s.origin_country_id = c1.id
         JOIN countries c2 ON s.destination_country_id = c2.id
         JOIN package_types pt ON s.package_type_id = pt.id
         JOIN shipping_modes sm ON s.shipping_mode_id = sm.id
WHERE s.polica = 'TRXT1234';
